Query al database


SELECT * FROM Post p JOIN Autore a
ON p.autore = a.id
WHERE p.id = 25;
                        

SELECT * FROM Post p JOIN Commento c
ON c.post_id = p.id
WHERE p.id = 25;
                        

SELECT * FROM Post p JOIN Tag_post tp
ON p.id = tp.post_id JOIN Tag t ON t.id = tp.tag_id
WHERE p.id = 25;
